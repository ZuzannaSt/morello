namespace AppBundle\Repository;

use Doctrine\ORM\EntityRepository;

/**
 * Class TaskRepository
 * @package AppBundle\Repository
 */
class TaskRepository extends EntityRepository
{
    /**
     * Find all objects ordered by name
     *
     * @return result
     */
    public function findAllOrderedByName()
    {
        return $this->getEntityManager()
            ->createQuery('
                SELECT t
                FROM AppBundle:Task t
                ORDER BY t.name ASC
            ')
            ->getResult();
    }

    /**
     * Add entity
     *
     * @param task
     * @param board_id
     * @return entity
     */
    public function add($task, $board_id)
    {
        $board = $this->getEntityManager()
          ->getRepository('AppBundle:Board')
          ->findOneById($board_id);

        $task->setCreatedAt(new \DateTime(date('Y-m-d H:i:s')));
        $task->setBoard($board);
        $this->save($task);
    }

    /**
     * Save entity
     *
     * @param task
     * @return entity
     */
    public function save($task)
    {
        $em = $this->getEntityManager();
        $em->persist($task);
        $em->flush();
    }

    /**
     * Delete entity
     *
     * @param task
     * @return entity
     */
    public function delete($task)
    {
        $em = $this->getEntityManager();
        $em->remove($task);
        $em->flush();
    }
}
